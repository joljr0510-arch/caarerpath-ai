<script>
function login(){
    let nama=document.getElementById("nama").value;
    if(nama==""){
        alert("Masukkan nama dulu!");
        return;
    }
    document.getElementById("loginPage").classList.add("hidden");
    document.getElementById("testPage").classList.remove("hidden");
}

function analisis(){
    let mtk=parseInt(document.getElementById("mtk").value);
    let ipa=parseInt(document.getElementById("ipa").value);
    let ips=parseInt(document.getElementById("ips").value);
    let bhs=parseInt(document.getElementById("bhs").value);
    let minat=document.querySelector('input[name="minat"]:checked');

    if(!mtk||!ipa||!ips||!bhs||!minat){
        alert("Lengkapi semua data!");
        return;
    }

    let ipaScore=(mtk+ipa)/2;
    let ipsScore=(ips+bhs)/2;

    let rekomendasi="";
    let penjelasan="";

    if(ipaScore>ipsScore && minat.value=="IPA"){
        rekomendasi="Jurusan IPA";
        penjelasan="Nilai Matematika dan IPA kamu kuat serta minat di bidang sains. Kamu cocok di jurusan IPA.";
    }
    else if(ipsScore>ipaScore && minat.value=="IPS"){
        rekomendasi="Jurusan IPS";
        penjelasan="Kekuatan kamu ada di bidang sosial dan bahasa. Jurusan IPS lebih sesuai.";
    }
    else{
        rekomendasi="Jurusan Bahasa / Konsultasi BK";
        penjelasan="Nilai dan minat kamu cukup seimbang. Disarankan konsultasi dengan guru BK untuk keputusan terbaik.";
    }

    document.getElementById("hasil").innerHTML=
    "<h3>🎯 Rekomendasi: "+rekomendasi+"</h3><p>"+penjelasan+"</p>";

    tampilChart(mtk,ipa,ips,bhs);
}

function tampilChart(mtk,ipa,ips,bhs){
    document.getElementById("chart").classList.remove("hidden");

    new Chart(document.getElementById("chart"),{
        type:'bar',
        data:{
            labels:['Matematika','IPA','IPS','Bahasa'],
            datasets:[{
                label:'Nilai Kamu',
                data:[mtk,ipa,ips,bhs]
            }]
        }
    });
}
